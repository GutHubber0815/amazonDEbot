<script>

    const do_checkout=true;


    const href=window.location.href.toLowerCase();


    function on_elem_present(selector,check_every=100) {

        return new Promise(function(resolve,reject) {

            const check_interval=setInterval(function() {

                if(document.querySelector(selector)) {

                    clearInterval(check_interval);

                    return resolve();

                }

            },check_every);

        });

    }


    function sleep(i) {

        return new Promise(function(resolve,reject) {

            setTimeout(resolve,i);

        });

    }


    function add_to_cart() {

        const s='.a-button-input';

        on_elem_present(s)

            .then(sleep.bind(100))

            .then(document.querySelector(s).click());

    }


    function place_order() {

        const s='input[name="placeYourOrder1"]'

        on_elem_present(s)

            .then(sleep.bind(100))

            .then(document.querySelector(s).click());

    }


    if(href.match('https://www.amazon.de/gp/aws/cart/.*'))

        add_to_cart();

    else if(do_checkout&&href.match('https://www.amazon.de/gp/buy/spc/.*'))

        place_order();

</script>
